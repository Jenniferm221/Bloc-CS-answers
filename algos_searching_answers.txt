are in the letter M, so you grab a handful more pages and open again and lets say you are in the letter S, so then you need to go back some pages until you are in

1. You and a friend have set a wager to see who can find the word "Albatross" in the dictionary the fastest. Write a program to allow you to win the bet.

A. 
let dictionary = [ "Accent","Ace","Albitros", "anotherTest", "random", "test"]


function binarySearch(dictionary, word){
  
let start = 0;
 
let end = dictionary.length - 1;
  
let middle = Math.floor((start + end) / 2);

  

while(dictionary[middle] !== word && start <= end){
    
if(dictionary[middle] < word) {
      
start = middle + 1;
    
} else { 
      
end = middle - 1;
   
} middle = Math.floor((start + end) / 2);
  
}
  
return dictionary[middle] === word ? middle : -1;

}


console.log(binarySearch(dictionary, 'Albitros'))
